# 4. Skill Extraction Function (Dictionary + Fuzzy + Substring)
def extract_skills_dict(text, skills_dict, fuzzy_threshold=90, min_word_len=2):
    text = text.lower()
    found_skills = set()
    # Exact and substring match
    for skill in skills_dict:
        if skill in text:
            found_skills.add(skill)
    # Fuzzy match for skills not already found
    words = set(w for w in text.split() if len(w) >= min_word_len)
    for skill in skills_dict - found_skills:
        best, score, _ = process.extractOne(skill, words, scorer=fuzz.ratio) if words else (None, 0, None)
        if score >= fuzzy_threshold:
            found_skills.add(skill)
    return list(found_skills)

# 7. Skill2Vec Embeddings (with fallback for OOV/multi-word)
def skill2vec_embed_fallback(skills):
    vectors = []
    for skill in skills:
        skill_norm = skill.strip().lower()
        if skill_norm in skill2vec.wv:
            vectors.append(skill2vec.wv[skill_norm])
        else:
            words = skill_norm.split()
            word_vecs = [skill2vec.wv[w] for w in words if w in skill2vec.wv]
            if word_vecs:
                vectors.append(np.mean(word_vecs, axis=0))
    if vectors:
        return np.mean(vectors, axis=0)
    else:
        return np.zeros(skill2vec.vector_size)

df['resume_skill2vec'] = df['resume_skills'].progress_apply(skill2vec_embed_fallback)
df['job_skill2vec'] = df['job_skills'].progress_apply(skill2vec_embed_fallback)

import pandas as pd
from IPython.display import display, Markdown

# Load the skills dictionary (should already be loaded in your notebook)
skills_file = 'skills_master_list.txt'
with open(skills_file, 'r', encoding='utf-8') as f:
    skills_dict = set(line.strip().lower() for line in f if line.strip() and not line.startswith('#'))

# Define unlabeled resume-job pairs
unlabeled_pairs = [
    {"resume": 'Senior Java developer with experience in Spring Boot and cloud deployments.',
     "job": 'Looking for a backend engineer skilled in Java and AWS cloud.'},
    {"resume": 'Marketing specialist with a focus on social media campaigns and brand management.',
     "job": 'Seeking a digital marketer to manage social media and branding.'},
    {"resume": 'Entry-level data analyst familiar with Excel and Tableau.',
     "job": 'Hiring a senior data scientist with Python and machine learning expertise.'},
    {"resume": 'Full-stack developer: React, Node.js, MongoDB.',
     "job": 'Frontend developer needed for React-based web apps.'},
    {"resume": 'Experienced HR manager with recruitment and onboarding expertise.',
     "job": 'Looking for a talent acquisition specialist.'},
    {"resume": 'Experienced HR manager with recruitment and onboarding expertise.',
      "job": 'Frontend developer needed for React-based web apps.'},
     {"resume": 'Senior Java developer with experience in Spring Boot and cloud deployments.',
     "job": 'Hiring a senior data scientist with Python and machine learning expertise.'}
]

# Compute SBERT embeddings
resume_texts = [pair['resume'] for pair in unlabeled_pairs]
job_texts = [pair['job'] for pair in unlabeled_pairs]
resume_sbert = sbert_model.encode(resume_texts, show_progress_bar=False)
job_sbert = sbert_model.encode(job_texts, show_progress_bar=False)

# Extract skills using the new dictionary-based extractor
resume_skills = [extract_skills_dict(text, skills_dict) for text in resume_texts]
job_skills = [extract_skills_dict(text, skills_dict) for text in job_texts]

# Use the fallback Skill2Vec embedding function from the notebook
def skill2vec_embed_fallback(skills):
    vectors = []
    for skill in skills:
        skill_norm = skill.strip().lower()
        if skill_norm in skill2vec.wv:
            vectors.append(skill2vec.wv[skill_norm])
        else:
            words = skill_norm.split()
            word_vecs = [skill2vec.wv[w] for w in words if w in skill2vec.wv]
            if word_vecs:
                vectors.append(np.mean(word_vecs, axis=0))
    if vectors:
        return np.mean(vectors, axis=0)
    else:
        return np.zeros(skill2vec.vector_size)

# Compute Skill2Vec similarities using aggregated embeddings
resume_skill2vec = [skill2vec_embed_fallback(skills) for skills in resume_skills]
job_skill2vec = [skill2vec_embed_fallback(skills) for skills in job_skills]
from sklearn.metrics.pairwise import cosine_similarity
sbert_sims = [cosine_similarity([r], [j])[0][0] for r, j in zip(resume_sbert, job_sbert)]
skill2vec_sims = [cosine_similarity([r], [j])[0][0] for r, j in zip(resume_skill2vec, job_skill2vec)]
hybrid_scores = [0.8 * s + 0.2 * k for s, k in zip(sbert_sims, skill2vec_sims)]

# Display results in a formatted table
results = []
for i, pair in enumerate(unlabeled_pairs):
    results.append({
        'Pair': i+1,
        'Resume': pair['resume'],
        'Job': pair['job'],
        'SBERT similarity': f"{sbert_sims[i]:.3f}",
        'Skill2Vec similarity': f"{skill2vec_sims[i]:.3f}",
        'Hybrid score': f"{hybrid_scores[i]:.3f}"
    })
df_results = pd.DataFrame(results)[['Pair', 'Resume', 'Job', 'SBERT similarity', 'Skill2Vec similarity', 'Hybrid score']]

display(df_results.style.set_properties(**{'text-align': 'left'}).set_table_styles([{'selector': 'th', 'props': [('text-align', 'left')]}]))
for i, pair in enumerate(unlabeled_pairs):
    print(f"Pair {i + 1}:")

    print(f"  Extracted Resume Skills: {', '.join(resume_skills[i]) if resume_skills[i] else 'No skills extracted'}")
    print(f"  Job: {pair['job']}")
    print(f"  Extracted Job Skills: {', '.join(job_skills[i]) if job_skills[i] else 'No skills extracted'}")
